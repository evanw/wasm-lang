@import("imports.add")
def _add(x int, y int) int

@export("add")
def add(x int, y int) int {
  return _add(x, y)
}

@export("fib")
def fib(n int) int {
  if n < 2 { return n }
  return fib(n - 1) + fib(n - 2)
}

type List {
  Empty
  Link(head int, tail List)
}

@export("main")
def main() List {
  var list = Empty
  var i = 0

  while i < 10 {
    list = Link(i, list)
    i = i + 1
  }

  return list
}
